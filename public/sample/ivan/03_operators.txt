# Все эти операторы умеют работать со всеми скалярными типами,
# И умеют работать с литералами:
#    123, 100000        числовые литералы
#    'a', "a", '\n'     символьные литералы
#    true, false, eol   алиасы для 0, 1, '\n'

# причем, язык будет по-разному считать 5 * 5, a * 5 и a * b. Все для оптимизации


byte $a = 5 * 4 + 1;  # bf-у считать ничего не придется. Компилятор упростит выражение до a = 21
char $ch = "А" + $a;  # Получим 21-ую букву алфавита(ё идет нах$й)
bool $bool = ($ch >= 'А' && $ch <= 'Я') || true; # Тут упростится || true

out $a, eol, $ch, eol;
out $bool, eol, !$bool, eol;

# Это компилятор сократит до $a++, потому что остальное никуда не записывается
$a * $a++ / $ch > 12;

out $a++, eol;
out ++$a, eol;
