# Все эти операторы умеют работать со всеми скалярными типами,
# И умеют работать с литералами:
#   - 123, 100000        числовые литералы
#   - 'a', "a", '\n'     символьные литералы
#   - true, false, eol   специальные литералы

# причем, язык будет по-разному считать 5 * 5, a * 5 и a * b. Все для оптимизации


# Кстати, символьные и строковые литералы у меня различаются по длине,
#    а не по одинарным/двойным кавычкам, как в си
#    Со строками у меня умеют работать только операторы =(это дальше) и ,


byte $a = 5 * 4 + 1;  # bf считать ничего не придется. Компилятор упростит выражение до a = 26
char $ch = "А" + $a;  # Получим 21-ый символ из алфавита(ё идет нах$й)
bool $bool = ($ch >= 'А' && $ch <= 'Я') || true;

out $a, eol, $ch, eol;
out $bool, eol, !$bool, eol;

out $a++, eol;
out ++$a, eol;
