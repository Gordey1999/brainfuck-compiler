# title: Типы

# В моем языке всего 3 скалярных типа:

char a;
byte b; # почти как char, только вводится и выводится по-другому. Больше ничем не отличается
bool c;

# для ввода/вывода есть всего 2 команды:

in a, 'dummy', b, c;
out '\nsaved:\n', a, eol, b, eol, c;

# можно написать и in(a, b, c). Но все же это не функции
# команды in и out со всеми типами работают по разному:

# in <char> - читает один символ и сразу записывает его в переменную
#
# in <литерал> - читает один символ и сразу выбрасывает.
#   'dummy' выше нужен, чтобы можно было после символа(a) и до числа(b) ввести пробел или интер
#
# in <byte> - читает символы до интера или пробела. И преобразует в число.
#   Ожидает от тебя число от 0 до 255.
#   Если число будет больше 255, в переменную сохранится модуль этого числа (n % 256)
#   Если напишешь 'abcd', то ничего не поломается. Просто сохранится всякий хлам
#
# in <bool> - читает один символ. Если не '0', то сохраняется true


# out <литерал> можно передать false, true, 0, 10000, 'a', 'hello'
#   Все это преобразуется в строку. Поэтому и числа больше 255 можно указать
#   'eol' это просто алиас для '\n'
#
# out <char> Выводит значение переменной
#
# out <byte> Переводит значение в ячейке в число и выводит
#
# out <bool> Выводит бул в виде '0' или '1'
