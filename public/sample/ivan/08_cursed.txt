# Пример с прошлой прогой

char map[][] = [
  'a', 'b', 'v', 'g', 'd', 'e', 'zh', 'z', 'i', 'y',
  'k', 'l', 'm', 'n', 'o', 'p', 'r', 's', 't', 'u',
  'f', 'h', 'c', 'ch', 'sh', 'shch', '', 'y', '', 'e',
  'yu', 'ya'
]

char letter

do
{
  in letter

  if letter >= 'а' && letter <= 'я' out map[letter - 'а']

  else if letter >= 'А' && letter <= 'Я'
    for byte i = 0 i < sizeof map[0] && map[letter - 'А'][i] i++ {
      char ch = map[letter - 'А'][i] - 'a' + 'A'
      out ch
    }

  else out letter
}
while letter != eol
