# Память:
# 0 - делимое
# 1 - делитель
# 2 - частное
# 3 - остаток
# 4 - всякое
# 5 - всякое
# 6 - всякое

# заполняем делимое и делитель
-
>
+++
<

# цикл деления
[                                   # пока делимое больше нуля
  >>>[-]<<<                         # сбрасываем остаток
  [->>>+>+<<<<]>>>>[-<<<<+>>>>]<<<< # копируем делимое в остаток
  >                                 # переходим к делителю
  [->>>+>+<<<<]>>>>[-<<<<+>>>>]     # копируем делитель в 4 ячейку

  # нам нельзя просто вычесть делитель из делимого, так как может случиться переполнение
  # поэтому будем отнимать по одному и проверять, дошли до нуля или нет. Для этого нужно каждый раз копировать делимое

  <                                 # переходим к четвертой ячейке с делителем
  [                                 # пока она непустая
    -<<<<                           # отнимаем единицу и переходим к делимому
    [->>>>>+>+<<<<<<]>>>>>>[-<<<<<<+>>>>>>]<  # копируем делимое в 5 ячейку
    [<<<<<->>>>>[-]]                # если больше нуля, отнимаем 1 от делимого
    <                               # возвращаемся к копии делителя (4)
  ]
  <<+<<                             # увеличиваем частное и переходим к делимому
]

# Готово... почти
# Теперь у нас либо частное больше нужного на единицу(нецелочисленное деление)
# Либо остаток должен быть равен нулю, а он равен делителю

# исправляем
>>>                         # двигаемся к остатку
[->+>+<<]>>[-<<+>>]<        # копируем его в 4 ячейку
<<<[->>>-<<<]>>>            # вычитаем делитель из остатка. Если получился 0, то разделили без остатка
[->[-]+>[-]+<<]>>[-<<+>>]<< # делаем 2 одинаковых була в 4 и 5 ячейках: 1 - остаток есть, 0 - остатка нет
[-<<->>]                    # если остаток больше нуля, вычитаем 1 из частного
>-                          # переходим ко второму булу и применяем логическое отрицание
[<<[-]>>[-]]                # если остаток равен нулю, сбрасываем основной остаток
